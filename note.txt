# Open-TeleVision 저장소 클론
git clone https://github.com/OpenTeleVision/TeleVision.git
cd TeleVision

# ⚠️ 중요: ROS 2 환경을 먼저 로드해야 함
source /opt/ros/humble/setup.bash
source ~/ros2_ws/install/setup.bash  # Unitree 패키지가 있다면

# 시스템 패키지 접근 가능한 가상환경 생성
python3 -m venv --system-site-packages tv_env

# 가상환경 활성화 (프롬프트에 (tv_env) 표시됨)
source tv_env/bin/activate
